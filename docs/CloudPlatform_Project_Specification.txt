    Building an AWS-like Cloud Service Platform with Java Technologies

PROJECT OVERVIEW:
================================================================================
Project Name: MiniCloud - Enterprise Cloud Platform
Description: A comprehensive cloud computing platform similar to AWS that provides
infrastructure services including compute instances, object storage, managed
databases, load balancing, and monitoring capabilities. Built entirely using
Java-based technologies.

Target Users: Developers, Organizations, DevOps Teams
Core Value Proposition: Self-hosted, cost-effective alternative to AWS with
similar features and APIs


SYSTEM ARCHITECTURE:
================================================================================

1. THREE-TIER ARCHITECTURE:
   
   PRESENTATION LAYER:
   - JavaFX Desktop Application (Admin Console)
   - Vaadin Web Application (User Portal)
   - RESTful API Endpoints
   
   BUSINESS LOGIC LAYER:
   - Spring Boot Microservices
   - Service Orchestration
   - Business Rules Engine
   - Event Processing
   
   DATA LAYER:
   - MySQL Database (Relational Data)
   - PostgreSQL (Metadata & Configuration)
   - MongoDB (Logs & Analytics)
   - Redis (Caching & Sessions)
   - MinIO (Object Storage)


2. MICROSERVICES COMPONENTS:

   a) Identity & Access Management (IAM) Service
   b) Compute Service (Virtual Machine Management)
   c) Storage Service (S3-compatible Object Storage)
   d) Database Service (Managed Database Provisioning)
   e) Load Balancer Service
   f) Monitoring & Metrics Service
   g) Billing & Metering Service
   h) API Gateway Service
   i) Notification Service
   j) Backup & Recovery Service

-- Users Table (MySQL)
CREATE TABLE users (
    user_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    tenant_id BIGINT NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    status ENUM('active', 'inactive', 'suspended') DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    last_login TIMESTAMP NULL,
    FOREIGN KEY (tenant_id) REFERENCES tenants(tenant_id),
    INDEX idx_tenant (tenant_id),
    INDEX idx_email (email)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Instances Table (PostgreSQL)
CREATE TABLE instances (
    instance_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id BIGINT NOT NULL,
    name VARCHAR(100) NOT NULL,
    instance_type VARCHAR(50) NOT NULL,
    image_id UUID NOT NULL,
    state VARCHAR(20) NOT NULL DEFAULT 'pending',
    private_ip INET,
    public_ip INET,
    security_group_id UUID,
    ssh_key_id UUID,
    vcpu INTEGER NOT NULL,
    memory_gb INTEGER NOT NULL,
    disk_gb INTEGER NOT NULL,
    region VARCHAR(50) NOT NULL,
    availability_zone VARCHAR(50),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    terminated_at TIMESTAMP WITH TIME ZONE,
    FOREIGN KEY (image_id) REFERENCES images(image_id),
    FOREIGN KEY (security_group_id) REFERENCES security_groups(group_id),
    FOREIGN KEY (ssh_key_id) REFERENCES ssh_keys(key_id)
);

CREATE INDEX idx_instances_tenant ON instances(tenant_id);
CREATE INDEX idx_instances_state ON instances(state);
CREATE INDEX idx_instances_created ON instances(created_at);

-- Buckets Table (MySQL)
CREATE TABLE buckets (
    bucket_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    tenant_id BIGINT NOT NULL,
    name VARCHAR(255) UNIQUE NOT NULL,
    region VARCHAR(50) NOT NULL,
    versioning_enabled BOOLEAN DEFAULT FALSE,
    encryption_enabled BOOLEAN DEFAULT FALSE,
    public_access_blocked BOOLEAN DEFAULT TRUE,
    storage_class VARCHAR(50) DEFAULT 'STANDARD',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (tenant_id) REFERENCES tenants(tenant_id),
    INDEX idx_bucket_tenant (tenant_id),
    INDEX idx_bucket_name (name)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Invoices Table (MySQL)
CREATE TABLE invoices (
    invoice_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    tenant_id BIGINT NOT NULL,
    invoice_number VARCHAR(50) UNIQUE NOT NULL,
    billing_period_start DATE NOT NULL,
    billing_period_end DATE NOT NULL,
    subtotal DECIMAL(12,2) NOT NULL,
    tax_amount DECIMAL(12,2) DEFAULT 0.00,
    total_amount DECIMAL(12,2) NOT NULL,
    currency VARCHAR(3) DEFAULT 'USD',
    status ENUM('draft', 'sent', 'paid', 'overdue', 'cancelled') DEFAULT 'draft',
    due_date DATE NOT NULL,
    paid_date TIMESTAMP NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (tenant_id) REFERENCES tenants(tenant_id),
    INDEX idx_invoice_tenant (tenant_id),
    INDEX idx_invoice_status (status),
    INDEX idx_invoice_due_date (due_date)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Metrics Table (PostgreSQL with TimescaleDB)
CREATE TABLE metrics (
    metric_id BIGSERIAL PRIMARY KEY,
    timestamp TIMESTAMP WITH TIME ZONE NOT NULL,
    resource_type VARCHAR(50) NOT NULL,
    resource_id VARCHAR(255) NOT NULL,
    metric_name VARCHAR(100) NOT NULL,
    value DOUBLE PRECISION NOT NULL,
    unit VARCHAR(20),
    dimensions JSONB,
    tenant_id BIGINT NOT NULL
);

CREATE INDEX idx_metrics_resource ON metrics(resource_type, resource_id);
CREATE INDEX idx_metrics_timestamp ON metrics(timestamp DESC);
CREATE INDEX idx_metrics_tenant ON metrics(tenant_id);
